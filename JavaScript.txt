//JavaScript
//syntax in javascript
//let, const, var, are used to declare variables in javascript
let a =10
//let is block scoped, cannot be redeclared in the same scope
const b=10; 
//const is block scoped and cannot be reassigned
var c = 30;
//var is function scoped cannot be redeclared in the same scope
console.log(a,b,c);
//console.log will give the  output of the values a,b,and c
//Output: 10,20,30
